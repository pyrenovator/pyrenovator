def trans_normalize(pc, feat, t_normalize):
    dim = t_normalize.get("recentering", [0, 1, 2])
    pc[:, dim] = pc[:, dim] - pc.mean(0)[dim]

    if t_normalize.get("method", None):
        method = t_normalize["method"]
