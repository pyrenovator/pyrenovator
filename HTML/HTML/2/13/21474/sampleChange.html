def uniform_init(module, a=0, b=1, bias=0):
    if getattr(module, "weight", None) is not None:
        nn.init.uniform_(module.weight, a, b)
    if getattr(module, "bias", None) is not None:
        nn.init.constant_(module.bias, bias)
